// --- ZMIENNE GLOBALNE SZYMON ---
bool flaga_BlokadaDrzwi = false;  
bool stan_PoLosowaniu = false;    

// --- FUNKCJE ---
void konfigurujLosomat() {
  randomSeed(analogRead(0)); 
}

void sprawdzLosomat() {
  int rzutMoneta = random(0, 2); 
  if (rzutMoneta == 0) {
    flaga_BlokadaDrzwi = false;
    Serial.println("WYRZUCONO: 0 -> Czysto. Otwieram drzwi (Blokada = FALSE).");
  } else {
    flaga_BlokadaDrzwi = true;
    Serial.println("WYRZUCONO: 1 -> PRZESZUKANIE! (Blokada = TRUE).");
  }
  stan_PoLosowaniu = true; 
}

void zresetujBlokade() {
  flaga_BlokadaDrzwi = false;
  stan_PoLosowaniu = false;
  Serial.println("-> Ochrona zresetowala blokade.");
}

// 
// KOD TESTOWY (SYMULACJA DZIAŁANIA)
// 
void setup() {
  Serial.begin(9600); // Uruchomienie monitora portu szeregowego
  Serial.println("Start testu losomatu...");
  konfigurujLosomat();
}

void loop() {
  Serial.println("\n--- ktos naciska przycisk wyjscia ---");
  sprawdzLosomat(); // Wywołanie logiki Konrada
  
  delay(2000); // Symulacja czasu reakcji (2 sekundy)
  
  // Jeśli wylosowano przeszukanie, symulujemy reset po chwili
  if(flaga_BlokadaDrzwi == true) {
    Serial.println("-> Czekamy na interwencje ochrony...");
    delay(3000); // 3 sekundy stania w zablokowanych drzwiach
    zresetujBlokade();
  }
  
  delay(3000); // Przerwa 3 sekundy, zanim podejdzie kolejna osoba
}
